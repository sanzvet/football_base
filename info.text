
# Создай виртуальное окружение
python3 -m venv .venv

# Активируй окружение (macOS/Linux)
source .venv/bin/activate

# Обновим pip внутри venv
pip install --upgrade pip

# Установи все зависимости из requirements.txt
pip install -r requirements.txt

# Установи браузеры для Playwright (критично для парсера!)
playwright install
